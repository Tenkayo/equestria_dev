###Rosa Redeemed
country_event = {
    id = arcturian.6
    title = arcturian.6.t
    desc = arcturian.6.d
    picture = GFX_badlands

    fire_only_once = yes
	is_triggered_only = yes

	option = {
		name = arcturian.6.a
		
		if = {
			limit = {
				has_country_flag = DED_Went_Blood
			}
			create_corps_commander = {
				name = "Rosa Maladicta Redeemed Vampire"
				portrait_path = "gfx/leaders/DED/Rosa_vampire.tga"
				traits = { redeemed_queen  }
				skill = 3
				attack_skill = 5
				defense_skill = 2
				planning_skill = 3
				logistics_skill = 2
			}
			else = {
				create_corps_commander = {
					name = "Rosa Maladicta Redeemed Lich"
					portrait_path = "gfx/leaders/DED/Rosa_lich.tga"
					traits = { redeemed_queen  }
					skill = 3
					attack_skill = 5
					defense_skill = 2
					planning_skill = 3
					logistics_skill = 2
				}
			}
		}

		
		#william must be in charge so fire the shipping sub plot
		if = {
			limit = {
				HLR = {
					has_government = democratic
				}
			}
			country_event = { id = arcturian.7 days = 60 }
		}
		
	}

}
