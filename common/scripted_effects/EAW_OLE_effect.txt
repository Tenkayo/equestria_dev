OLE_CHN_intervention = {
	random_country = {
		limit = {
			original_tag = OLE
			NOT = { has_government = neutrality }
		}
		CHN = { declare_war_on = { target = PREV type = annex_everything } }
	}
}

#Scope is CHN
OLE_CHN_intervention_step2 = {
	random_country = {
		limit = {
			original_tag = OLE
			NOT = { is_subject_of = CHN }
		}
		random_country = {
			limit = {
				original_tag = OLE
				is_subject_of = CHN
			}
			add_to_war = { targeted_alliance = CHN enemy = PREV }
		}
	}
}
